%% 1. Motivación: El Problema del Patch Testing
\begin{frame}{Motivación: El Problema del Patch Testing}
\begin{itemize}
    \item Los cambios en el software suelen introducir errores y vulnerabilidades.
    \item La cobertura de lineas es insuficiente: se puede cubrir un parche sin ejecutar el \textbf{nuevo comportamiento} introducido.
    \item \textbf{Limitación actual:} Los casos de prueba a menudo ejercitan el mismo comportamiento en ambas versiones.
    \item \textbf{Solución:} Generar entradas que fuercen la divergencia entre la versión antigua y la nueva.
\end{itemize}
\end{frame}\

%% 3. Shadow Intro
\begin{frame}{Shadow}
\begin{itemize}
  \item \textbf{¿Qué hace?}
  \begin{itemize}
    \item Ejecuta ambas versiones en la misma instancia de Ejecución Simbólica.
    \item Determina precisamente cuando hay divergencia en el comportamiento
    \item Tiempo de ejecución y consumo de memoria bajos
  \end{itemize}
  \item \textbf{¿Cómo lo hace?}
  \begin{enumerate}
    \item Unifica ambas versiones usando anotaciones \texttt{change()}.
    \item Selecciona casos de prueba que ejecuten el parche.
    \item Symboluc Execution con bifurcación de cuatro vías para detectar divergencias.
    \item Re-ejecución nativa para validar divergencias y clasificar resultados.
  \end{enumerate}
\end{itemize}
  \end{frame}

    \draw [arrow] (old) -- (unify); \draw [arrow] (new) -- (unify);
    \draw [arrow] (unify) -- (select); \draw [arrow] (select) -- (symexec);
    \draw [arrow] (symexec) -- (replay); \draw [arrow] (replay) -- (res);
\end{tikzpicture}
\end{frame}


% 5. Paso 1: Unificación de Versiones
\begin{frame}{Unificación de Versiones: El Macro \texttt{change()}}
\begin{itemize}
    \item Las versiones se ejecutan en \textbf{lockstep} (sincronía) hasta divergir.
    \item \textbf{Patrones de Anotación:}
    \begin{itemize}
        \item \textbf{Rvalues:} \texttt{x = change(E1, E2)}
        \item \textbf{Condicionales:} \texttt{if (A \&\& change(true, B))} para fortalecer condiciones.
        \item \textbf{Código Lineal:} \texttt{if (change(true, false)) \{ old\_code \}} para eliminar fragmentos.
    \end{itemize}
\end{itemize}

\begin{frame}{Bifurcación de Cuatro Vías (Four-way Forking)}
\centering
\begin{tikzpicture}[level distance=1.2cm,
  level 1/.style={sibling distance=2.5cm},
  level 2/.style={sibling distance=1.2cm},
  scale=0.7, every node/.style={scale=0.7}]
  
  \node [circle, draw] {Branch}
    child {node [rectangle, draw] {$new \land old$}
      child {node {Same Path}}
    }
    child {node [rectangle, draw, fill=gray!30] {$new \land \neg old$}
      child {node {Diff (Divergencia)}}
    }
    child {node [rectangle, draw, fill=gray!30] {$\neg new \land old$}
      child {node {Diff (Divergencia)}}
    }
    child {node [rectangle, draw] {$\neg new \land \neg old$}
      child {node {Same Path}}
    };
\end{tikzpicture}
\begin{itemize}
    \item Evalúa la condición de bifurcación bajo los estados de ambas versiones.
    \item Los caminos \textbf{Diff} (sombreados) son los que exponen comportamientos únicos del parche.
\end{itemize}
\end{frame}

\begin{frame}{Eficiencia: Shadow Expressions}
\centering
\begin{tikzpicture}[sibling distance=1.5cm, level distance=1cm, 
    every node/.style={circle, draw, minimum size=0.6cm}, scale=0.8]
    \node {z}
        child {node {y}
            child {node [rectangle, fill=yellow!20, minimum width=1.5cm] {SHADOW}
                child {node {a}}
                child {node {b}}
            }
            child {node {1}}
        }
        child {node {2}};
\end{tikzpicture}
\begin{itemize}
    \item Las expresiones que contienen cambios se mantienen como nodos \textbf{SHADOW}.
    \item \textbf{Ventajas:} Propagación implícita de cambios, creación en tiempo constante y máxima compartición de memoria.
    \item Si $e_{old} \equiv e_{new}$ bajo el PC actual, se simplifica la expresión.
\end{itemize}
\end{frame}

\begin{frame}{Fases de Exploración Dinámica}
\textbf{1. Fase Concolic:}
\begin{itemize}
    \item Ejecuta la entrada inicial. Si detecta que una divergencia es \textbf{posible} (aunque no se siga), genera un caso de prueba y lo encola.
\end{itemize}
\textbf{2. Bounded Symbolic Execution (BSE):}
\begin{itemize}
    \item Retoma los puntos de divergencia de la cola.
    \item Ejecuta una búsqueda en anchura (BFS) \textbf{solo en la versión nueva} para explorar el impacto del parche sistemáticamente.
\end{itemize}
\end{frame}

\begin{frame}{Oráculos y Cross-version Checks}
\begin{itemize}
    \item \textbf{Comparación de Salida:} Se contrastan \texttt{stdout} y \texttt{exit codes} en ejecución nativa.
    \item \textbf{Detección de Errores de Memoria:} Se utiliza \textbf{AddressSanitizer} para identificar desbordamientos de búfer o \textit{use-after-free} que no causan crash inmediato.
    \item \textbf{Clasificación automática:}
    \begin{itemize}
        \item Bug de regresión si la nueva falla y la antigua no.
        \item Divergencia esperada si el cambio es intencional según el commit.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Resultados en Coreutils}
\begin{itemize}
    \item Evaluado en 18 parches complejos de COREBench.
    \item \textbf{Caso de éxito (cut):} Encontró un desbordamiento de búfer en parches que gestionan rangos abiertos/cerrados.
    \item Expuso errores de tipo \textit{abort} y regresiones semánticas que pasaron desapercibidas en las suites de prueba originales.
    \item \textbf{Conclusión:} SSE reduce drásticamente el espacio de búsqueda al enfocarse solo en las divergencias.
\end{itemize}
\end{frame}
